            databaseURL: "https://dht22-b2230-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "dht22-b2230",
            storageBucket: "dht22-b2230.appspot.com",
            messagingSenderId: "774855203635",
            appId: "1:774855203635:web:6f43abc6455b67c8607c46" // Jalankan Langkah 3 jika belum tahu cara mendapatkan App ID
        };

        // Inisialisasi Firebase Koneksi
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();

        // Wadah penampung data untuk file Excel
        let arrayUntukExcel = [];

        // Mengambil data secara Real-time dari node 'DATA_SENSOR'
        database.ref('DATA_SENSOR').on('value', (snapshot) => {
            const data = snapshot.val();
            if (data) {
                const suhu = data.SUHU || 0;
                const kelembapan = data.KELEMBAPAN || 0;
                
                // Membuat format tanggal dan waktu lokal saat data tiba di laptop
                const waktuSekarang = new Date().toLocaleString('id-ID');

                // 1. Perbarui nilai angka besar pada Card secara instan
                document.getElementById('txt-suhu').innerHTML = `${suhu.toFixed(2)}<span class="fs-2">°C</span>`;
                document.getElementById('txt-kelembapan').innerHTML = `${kelembapan.toFixed(2)}<span class="fs-2">%</span>`;

                // 2. Hapus teks "Menunggu data..." di baris pertama tabel
                const loadingRow = document.getElementById('loading-row');
                if (loadingRow) loadingRow.remove();

                // 3. Simpan data ke memori lokal untuk siap diunduh ke Excel
                arrayUntukExcel.unshift({
                    "Waktu Pengambilan": waktuSekarang,
                    "Suhu (C)": suhu,
                    "Kelembapan (%)": kelembapan
                });

                // 4. Tambahkan baris baru ke dalam tabel HTML (di posisi teratas)
                const tbody = document.getElementById('data-rows');
                const rowBaru = `
                    <tr>
                        <td><strong>${waktuSekarang}</strong></td>
                        <td><span class="badge bg-danger fs-6">${suhu.toFixed(2)} °C</span></td>
                        <td><span class="badge bg-info text-dark fs-6">${kelembapan.toFixed(2)} %</span></td>
                    </tr>
                `;
                tbody.insertAdjacentHTML('afterbegin', rowBaru);
            }
        });

        // Fungsi Ekspor ke File Excel (.xlsx)
        function unduhExcel() {
            if (arrayUntukExcel.length === 0) {
                alert("Belum ada data sensor yang masuk. Pastikan ESP32 Anda menyala!");
                return;
            }

            // Proses konversi data JSON menjadi Worksheet Excel
            const worksheet = XLSX.utils.json_to_sheet(arrayUntukExcel);
            const workbook = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(workbook, worksheet, "Laporan DHT22");
            
            // Perintah bagi browser untuk mengunduh otomatis file Excel
            XLSX.writeFile(workbook, "Laporan_Data_Sensor_DHT22.xlsx");
        }
